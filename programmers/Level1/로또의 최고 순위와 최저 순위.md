### Solution
```javascript
function solution(lottos, win_nums) {
  let matchingCount = 0;
  let erasedNumber = 0;

  lottos.forEach(lotto => {
    if(lotto === 0) erasedNumber++;
    if(win_nums.includes(lotto)) {
      matchingCount++;
    }
  })

  let min = 7 - matchingCount == 7 ? 6 : 7 - matchingCount; 
  let max = min - erasedNumber < 1 ? 1 : min - erasedNumber;

  return [max, min]
}
```
