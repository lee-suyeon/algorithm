### Solution

```javascript
function solution(s) {
  let answer = s;
  const numbers = ["zero", "one", "two", "three", "four", "five", "six", "seven", "eight", "nine"];
  
  for(let i = 0; i < numbers.length; i++) {
    if(s.includes(numbers[i])) {
      let arr = answer.split(numbers[i]);
      answer = arr.join(i);
    }
  }

  return Number(answer);
}
```
