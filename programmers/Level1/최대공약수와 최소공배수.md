### Solution

```javascript
function solution(n, m) {
  let gcd = getGCD(n, m);
  let lcm = (n * m) / gcd;

  function getGCD(n, m) {
    if (m === 0) return n;
    return n > m ? getGCD(m, n % m) : getGCD(n, m % n);
  }

  return [gcd, lcm];
}
```
