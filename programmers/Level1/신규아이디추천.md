function solution1(new_id) {
  let answer = new_id
    .toLowerCase()
    .replace(/[^\w-._]/g, "")
    .replace(/\.{2,}/g, ".")
    .replace(/^\.|\.$/g, "")
    .padEnd(1, "a")
    .slice(0, 15)
    .replace(/^\.|\.$/g, "");

  while (answer.length < 3) {
    answer += answer[answer.length - 1];
  }
  return answer;
}
