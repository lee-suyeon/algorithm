const input = fs.readFileSync("/dev/stdin").toString().trim().split(" ");

const [n, k] = input.map((v) => Number(v));

let queue = Array(n)
  .fill()
  .map((v, i) => i + 1);
let answer = [];
let current = 1;

while (queue.length) {
  const shiftNum = queue.shift();
  if (current % k === 0) {
    answer.push(shiftNum);
  } else {
    queue.push(shiftNum);
  }
  current += 1;
}
